19uint16_t CmdParser::parseCmd(uint8_t *buffer, size_t bufferSize)
20{
21 bool isString = false;
22
23 // init param count
24 m_paramCount = 0;
25
26 // buffer is not okay
27 if (buffer == NULL || bufferSize == 0 || buffer[0] == 0x00) {
28 return CMDPARSER_ERROR;
29 }
30
31 // init buffer
32 m_buffer = buffer;
33 m_bufferSize = bufferSize;
34
36 // Run Parser
37 for (size_t i = 0; i < bufferSize; i++) {
38
39 // end
40 if (buffer[i] == 0x00 || m_paramCount == 0xFFFE) {
41 if (i > 0 && buffer[i - 1] != 0x00) {
42 m_paramCount++;
43 }
44 return m_paramCount;
45 }
46 // is string "xy zyx" / only the quote option is disabled
47 else if (!m_ignoreQuote && buffer[i] == CMDPARSER_CHAR_DQ) {
48 buffer[i] = 0x00;
49 isString = !isString;
50 }
51 // replace seperator with '\0'
52 else if (!isString && buffer[i] == m_seperator) {
53 buffer[i] = 0x00;
54 }
55 // replace = with '\0' if KEY=Value is set
56 else if (!isString && m_useKeyValue && buffer[i] == CMDPARSER_CHAR_EQ) {
57 buffer[i] = 0x00;
58 }
59
60 // count
61 if (i > 0 && buffer[i] == 0x00 && buffer[i - 1] != 0x00) {
62 m_paramCount++;
63 }
64 }
65
66 return m_paramCount;
67}

69char *CmdParser::getCmdParam(uint16_t idx)
70{
71 uint16_t count = 0;
72
73 // idx bigger than exists param
74 if (idx > m_paramCount) {
75 return NULL;
76 }
77
78 // search hole cmd buffer
79 for (size_t i = 0; i < m_bufferSize; i++) {
80
81 // find next position
82 if (i > 0 && m_buffer[i] == 0x00 && m_buffer[i - 1] != 0x00) {
83 count++;
84 }
85
86 // found indx with next character
87 if (count == idx && m_buffer[i] != 0x00) {
88 return reinterpret_cast<char *>(&m_buffer[i]);
89 }
90 }
91
92 return NULL;
93}

95char *CmdParser::getValueFromKey(const char *key, bool progmem)
96{
97 bool foundKey = false;
98
99 for (size_t i = 0; i < m_bufferSize; i++) {
100
101 // find data
102 if (m_buffer[i] == 0x00) {
103 continue;
104 }
105
106 // find first element
107 if (i == 0 || (m_buffer[i - 1] == 0x00 && !foundKey)) {
108 // if key in SRAM
109 if (!progmem &&
110 strcasecmp(reinterpret_cast<char *>(&m_buffer[i]), key) == 0) {
111 foundKey = true;
112 }
113 // if key in PROGMEM
114 else if (progmem &&
115 strcasecmp_P(reinterpret_cast<char *>(&m_buffer[i]),
116 key) == 0) {
117 foundKey = true;
118 }
119 }
120 // Next element after found key is the value...
121 else if (m_buffer[i - 1] == 0x00) {
122 return reinterpret_cast<char *>(&m_buffer[i]);
123 }
124 }
125
126 return NULL;
