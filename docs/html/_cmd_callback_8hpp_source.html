106template <size_t STORESIZE, typename T>
107class _CmdCallback : public CmdCallbackObject
108{
109 public:
113 _CmdCallback() : m_nextElement(0)
114 {
115 memset(m_cmdList, 0x00, sizeof(PGM_P) * STORESIZE);
116 memset(m_functList, 0x00, sizeof(CmdCallFunct) * STORESIZE);
117 }
118
126 bool addCmd(T cmdStr, CmdCallFunct cbFunct)
127 {
128 // Store is full
129 if (m_nextElement >= STORESIZE) {
130 return false;
131 }
132
133 // add to store
134 m_cmdList[m_nextElement] = cmdStr;
135 m_functList[m_nextElement] = cbFunct;
136
137 ++m_nextElement;
138 return true;
139 }
140
144 virtual size_t getStoreSize() { return STORESIZE; }
145
149 virtual bool checkStorePos(size_t idx)
150 {
151 if (idx < STORESIZE && m_cmdList[idx] != NULL) {
152 return true;
153 }
154
155 return false;
156 }
157
161 virtual bool callStoreFunct(size_t idx, CmdParser *cmdParser)
162 {
163 //auto (*fcnPtr)()= m_functList[idx](cmdParser);
164 if (idx < STORESIZE && m_functList[idx] != NULL) {
165 m_functList[idx](cmdParser);
166 return true;
167 }
168
169 return false;
170 }
171
172 protected:
174 T m_cmdList[STORESIZE];
175
177 CmdCallFunct m_functList[STORESIZE];
178
180 size_t m_nextElement;
181};
