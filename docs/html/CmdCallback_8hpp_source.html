125template <size_t STORESIZE, typename T>
126class _CmdCallback : public CmdCallbackObject
127{
128 public:
132 _CmdCallback() : m_nextElement(0)
133 {
134 memset(m_cmdList, 0x00, sizeof(PGM_P) * STORESIZE);
135 memset(m_functList, 0x00, sizeof(CmdCallFunct) * STORESIZE);
136 }
137
145 bool addCmd(T cmdStr, CmdCallFunct cbFunct)
146 {
147 // Store is full
148 if (m_nextElement >= STORESIZE) {
149 return false;
150 }
151
152 // add to store
153 m_cmdList[m_nextElement] = cmdStr;
154 m_functList[m_nextElement] = cbFunct;
155
156 ++m_nextElement;
157 return true;
158 }
159
163 virtual size_t getStoreSize() { return STORESIZE; }
164
168 virtual bool checkStorePos(size_t idx)
169 {
170 if (idx < STORESIZE && m_cmdList[idx] != NULL) {
171 return true;
172 }
173
174 return false;
175 }
176
180 virtual bool callStoreFunct(size_t idx, CmdParser *cmdParser)
181 {
182 //auto (*fcnPtr)()= m_functList[idx](cmdParser);
183 if (idx < STORESIZE && m_functList[idx] != NULL) {
184 current_idx=idx;
185 m_functList[idx](cmdParser);
186 }
187
188 return false;
189 }
190
191 protected:
193 T m_cmdList[STORESIZE];
194
196 CmdCallFunct m_functList[STORESIZE];
197
199 size_t m_nextElement;
200};
