2/*
3 a4988.cpp - - Arduino library for using the a4988 stepper driver
4 William Smith, 2014
5
6 The A4988 stepper driver is for Pololu stepper driver boards
7 and compatible clones. These boards use the Allegro a4988
8 stepper motor driver IC. (see Allegro website for datasheet)
9 This library diverges from others that are around, in that it
10 assumes that the MS1, MS2, and MS3 pins are connected to gpio
11 pins on the Arduino, allowing control over the microstepping
12 modes.
13
14 The A4988 is capable of microstepping down to 1/16 of a step,
15 enabling fine control over the stepper motor. This fine control
16 can be used in, among other things, 3D printers.
17 This library provides an interface for setting the different
18 step modes, going from full step down to 1/16 step, using a
19 simple setter function, where the argument is 1,2,4,8, or 16.
20
21 MS1 MS2 MS3
22 -----------------
23 low low low Full step
24 high low low Half step
25 low high low 1/4 step
26 high high low 1/8 step
27
28 high high high 1/16
29 Note:
30 Lower delay values can be used in the microstepping mode.
31 Values as low as 25 usec can be used in the 1/16 mode
32 with some motors.
33*/

37// constructor to set up pins and initialize data
38a4988::a4988(int motor_steps, int ms1_pin, int ms2_pin, int ms3_pin, int dir_pin, int enable_pin, int step_pin, bool direction)
39{
40 this->enable_pin = enable_pin;
41 this->dir_pin = dir_pin;
42 this->ms1_pin = ms1_pin;
43 this->ms2_pin = ms2_pin;
44 this->ms3_pin = ms3_pin;
45 this->step_pin = step_pin;
46 this->direction=direction;
47 // setup the pins on the microcontroller:
48 pinMode(this->ms1_pin, OUTPUT);
49 pinMode(this->ms2_pin, OUTPUT);
50 pinMode(this->ms3_pin, OUTPUT);
51 pinMode(this->dir_pin, OUTPUT);
52 pinMode(this->enable_pin, OUTPUT);
53 pinMode(this->step_pin, OUTPUT);
54
55 if(motor_steps != 0)
56 {
57 this->motor_steps = motor_steps;
58 }
59 else
60 {
61 this->motor_steps = 200; // a common value for steppers
62 }
63
64 // use setDelay to change before stepping, otherwise default
65 this->step_delay = STEP_DELAY; // 20000 us (20 ms) as a default
66
67}

105// step only once
106void a4988::stepOnce(void)
107{
108 digitalWrite(step_pin, HIGH);
109 delayMicroseconds(this->step_delay);
110 digitalWrite(step_pin, LOW);
111 delayMicroseconds(this->step_delay);
112 if(this->direction==up || this->direction==clk) TotalStepCounted++;
113 if(this->direction==down || this->direction==cclk) TotalStepCounted--;
114}

126// set microstepping mode
127// (1 /) 1, 2, 4, 8, or 16
128void a4988::setStepMode(int stepMode)
129{
130 this->step_mode = stepMode;
131 switch (stepMode)
132 {
133 // MS1 MS2 MS3
134 case 16: // high high high
135 digitalWrite(ms1_pin, HIGH);
136 digitalWrite(ms2_pin, HIGH);
137 digitalWrite(ms3_pin, HIGH);
138 break;
139 case 8: // high high low
140 digitalWrite(ms1_pin, HIGH);
141 digitalWrite(ms2_pin, HIGH);
142 digitalWrite(ms3_pin, LOW);
143 break;
144 case 4: // low high low
145 digitalWrite(ms1_pin, LOW);
146 digitalWrite(ms2_pin, HIGH);
147 digitalWrite(ms3_pin, LOW);
148 break;
149 case 2: // high low low
150 digitalWrite(ms1_pin, HIGH);
151 digitalWrite(ms2_pin, LOW);
152 digitalWrite(ms3_pin, LOW);
153 break;
154 case 1: // low low low
155 digitalWrite(ms1_pin, LOW);
156 digitalWrite(ms2_pin, LOW);
157 digitalWrite(ms3_pin, LOW);
158 break;
159 }
160}
